
\usetikzlibrary{shapes.misc}

\newcommand\titlebar{%
\tikz[baseline,trim left=3.1cm,trim right=3cm] {
    \fill [colorSection!41] (2.5cm,-1ex) rectangle (\textwidth+3.1cm,2.5ex);
    \node [
        fill=colorSection,
        anchor= base east,
        rounded rectangle,
        minimum height=3.5ex] at (3cm,0) {
        \textbf{\arabic{chapter}.\thesection.}
    };
}%
}

\titleformat{\section}{\Large\bfseries\sffamily}{\titlebar}{0.1cm}{#1}
\renewcommand*{\thesection}{\arabic{section}}
