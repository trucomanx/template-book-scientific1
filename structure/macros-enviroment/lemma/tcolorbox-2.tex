%% Create a new counter that will follow tcolorbox's numbering
\newcounter{lemma}[section]
\renewcommand*{\thelemma}{\noexpand\thesection.\noexpand\arabic{lemma}}

\newtcolorbox[list inside=MisLemma,list type=MisLemma,auto counter,number within=section]{lemma}[1][]{
  enhanced jigsaw,
  colback=white,
  colframe=colorsystemdefault!80,
  coltitle=colorsystemdefault,
  fonttitle=\bfseries\sffamily,
  sharp corners,
  detach title,
  leftrule=24mm,
  underlay unbroken and first={\node[below,text=white,font=\sffamily\bfseries,align=center]
    at ([xshift=-12mm,yshift=-1mm]interior.north west) {LEMMA\\\thetcbcounter};},
  breakable,pad at break=1mm,
  title={#1},
  code={
  \ifdefempty{\tcbtitletext}
    {
    \tcbset{before upper={\addtocounter{lemma}{-1}\refstepcounter{lemma} \refstepcounter{lemma}}}
    }
    {
    \tcbset{before upper={\addtocounter{lemma}{-1}\refstepcounter{lemma} \tcbtitle\par\medskip\refstepcounter{lemma}}}
    }
  },
}
