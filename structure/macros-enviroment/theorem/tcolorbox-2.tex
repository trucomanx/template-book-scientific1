%% Create a new counter that will follow tcolorbox's numbering
\newcounter{theorem}[section]
\renewcommand*{\thetheorem}{\noexpand\thesection.\noexpand\arabic{theorem}}

\newtcolorbox[list inside=MisTheorem,list type=MisTheorem,auto counter,number within=section]{theorem}[1][]{
  enhanced jigsaw,
  colback=white,
  colframe=colorsystemdefault,
  coltitle=colorsystemdefault,
  fonttitle=\bfseries\sffamily,
  sharp corners,
  detach title,
  leftrule=24mm,
  underlay unbroken and first={\node[below,text=white,font=\sffamily\bfseries,align=center]
    at ([xshift=-12mm,yshift=-1mm]interior.north west) {TEOREMA\\\thetcbcounter};},
  breakable,pad at break=1mm,
  title={#1},
  code={
  \ifdefempty{\tcbtitletext}
    {
    \tcbset{before upper={\addtocounter{theorem}{-1}\refstepcounter{theorem} \refstepcounter{theorem}}}
    }
    {
    \tcbset{before upper={\addtocounter{theorem}{-1}\refstepcounter{theorem} \tcbtitle\par\medskip\refstepcounter{theorem}}}
    }
  },
}
